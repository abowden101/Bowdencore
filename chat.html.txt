    <div class="container">
        <div class="header">

        </div>

        <div class="mode-toggle">
            <div class="mode-btn active" id="general-btn" onclick="setMode('general')">General Assistant</div>
            <div class="mode-btn" id="ticket-btn" onclick="setMode('ticket')">Ticket Assistant</div>
        </div>

        <div class="messages" id="messages"></div>

        <div class="input-area">
            <input type="text" id="prompt" placeholder="Ask anything... (Ticket mode is great for client emails)">
            <button onclick="sendMessage()">Send</button>
        </div>
    </div>

    <script>
        let currentMode = "general";

        function setMode(mode) {
            currentMode = mode;
            document.querySelectorAll('.mode-btn').forEach(btn => btn.classList.remove('active'));
            document.getElementById(mode + '-btn').classList.add('active');
        }

        async function sendMessage() {
            const input = document.getElementById('prompt');
            const promptText = input.value.trim();
            if (!promptText) return;

            addMessage(promptText, 'user');
            input.value = '';

            const loading = addMessage("Analyzing ticket...", 'assistant');

            try {
                const res = await fetch('http://127.0.0.1:8000/mcp/query', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ 
                        prompt: promptText,
                        mode: currentMode 
                    })
                });

                const data = await res.json();
                loading.remove();

                if (data.status === "success") {
                    addMessage(data.response, 'assistant');
                } else {
                    addMessage("Error: " + (data.response || "Unknown error"), 'assistant');
                }
            } catch (e) {
                loading.remove();
                addMessage("Cannot connect to Local Grok MCP. Make sure mcp_server.py is running.", 'assistant');
            }
        }

        function addMessage(text, type) {
            const div = document.createElement('div');
            div.className = `message ${type}`;
            div.textContent = text;
            document.getElementById('messages').appendChild(div);
            div.scrollIntoView({ behavior: "smooth" });
            return div;
        }

        document.getElementById('prompt').addEventListener('keypress', e => {
            if (e.key === 'Enter') sendMessage();
        });
    </script>
